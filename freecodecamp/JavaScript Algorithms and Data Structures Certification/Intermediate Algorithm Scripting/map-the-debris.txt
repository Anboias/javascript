function orbitalPeriod(arr) {
  const GM = 398600.4418;
  const earthRadius = 6367.4447;

  let newArr = [...arr];

  for (let i of newArr) {
    i.orbitalPeriod = doMath(i.avgAlt);
    delete i.avgAlt;
  }
  function doMath(k) {
    return Math.round((2 * Math.PI * Math.sqrt((Math.pow(k + earthRadius, 3) / GM))));
 }
  return newArr;
}
